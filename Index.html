function calcular() { const comprimento = parseFloat(document.getElementById('comprimento').value); const espacamentoCm = parseFloat(document.getElementById('espacamento').value); const altura = parseFloat(document.getElementById('altura').value); const barra = parseFloat(document.getElementById('barra').value); const extremidades = document.getElementById('extremidades').checked; const espessuraBarra = parseFloat(document.getElementById('espessuraBarra').value) / 100; if (isNaN(comprimento) || isNaN(espacamentoCm) || isNaN(altura) || isNaN(barra) || isNaN(espessuraBarra)) { document.getElementById('resultado').innerHTML = 'Preencha todos os campos corretamente.'; return; } const alturaUtil = altura - (2 * espessuraBarra); if (alturaUtil <= 0) { document.getElementById('resultado').innerHTML = 'A espessura das barras é maior que a altura total!'; return; } const espacamento = espacamentoCm / 100; const vaos = comprimento / espacamento; let montantes = Math.floor(vaos); if (extremidades) { montantes += 1; } const pecasPorBarra = Math.floor(barra / alturaUtil); const barrasNecessarias = Math.ceil(montantes / pecasPorBarra); document.getElementById('resultado').innerHTML = ` Montantes verticais: ${montantes}
Altura útil dos montantes: ${alturaUtil.toFixed(3)} m
Peças por barra de ${barra} m: ${pecasPorBarra}
Barras necessárias: ${barrasNecessarias} `; }
